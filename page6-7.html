Завдання: Знижка на медикаменти

Напишіть функцію, яка приймає массив об'єктів і повертає новий масив об'єктів, у якому є знижка на 30% на всі медикаменти, ціна яких перевищує 300 грн. Надайте ID для кожного медикамента.

<script>
// Початковий масив медикаментів
const medicines = [
    { name: "Noshpa", price: 170 },
    { name: "Analgin", price: 55 },
    { name: "Quanil", price: 310 },
    { name: "Alphacholine", price: 390 },
];

// Функція для створення знижки
const applyDiscount = (medicines) => {
    return medicines.map((medicine, index) => {
        let discountPrice = medicine.price;
        if (medicine.price > 300) {
            discountPrice = medicine.price * 0.7;
        }
        return { id: index + 1, name: medicine.name, price: discountPrice.toFixed(2) };
    });
};

// Виконання функції
const discountedMedicines = applyDiscount(medicines);
console.log(discountedMedicines);

// Вивід результатів на сторінку
document.addEventListener('DOMContentLoaded', () => {
    const resultContainer = document.getElementById('result');
    resultContainer.innerHTML = '';
    discountedMedicines.forEach(medicine => {
        const medicineElement = document.createElement('div');
        medicineElement.textContent = `ID: ${medicine.id}, Назва: ${medicine.name}, Ціна зі знижкою: ${medicine.price} грн`;
        resultContainer.appendChild(medicineElement);
    });
});
</script>
